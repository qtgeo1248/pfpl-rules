\subsection{Statics}

\subsubsection{Expressions}
$\Gamma \entails \IsOf{e}{\tau}$ have the same rules as before.

\subsubsection{Stacks}

\boxed{\IsOfStack{k}{\tau}}
\begin{mathpar}

\Infer
  {\IsOfStack{k}{\tau}}
  {\IsOfStack{\ConsStack{\raiseabt{}{\text{--}}}{k}}{\exntyabt}}

\Infer
  {\IsOfStack{k}{\tau} \\ \IsOf{x}{\exntyabt} \entails \IsOf{e_1}{\tau}}
  {\IsOfStack{\ConsStack{\handleabt{\text{--}}{x}{e_1}}{k}}{\tau}}

\end{mathpar}

\end{mathpar}
